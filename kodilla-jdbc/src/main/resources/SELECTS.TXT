SELECT P.SUMMARY, IL.NAME FROM ISSUESS P, ISSUESLISTS IL WHERE P.ISSUESLIST_ID = IL.ID;

SELECT P.ID, P.SUMMARY, IL.NAME , U.FIRSTNAME, U.LASTNAME
FROM ISSUESS P, ISSUESLISTS IL, USERS U
WHERE P.ISSUESLIST_ID = IL.ID AND P.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS POSTS_COUNTER
FROM ISSUESS P, USERS U
WHERE P.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID;
